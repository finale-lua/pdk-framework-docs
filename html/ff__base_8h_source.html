1027 {
1028 if (!pCompareObject)
1029 {
1030 DebugOut("DebugDataCompare(): Compare object is nil.");
1031 return;
1032 }
1033 if (_GetLoadedSize() != pCompareObject->_GetLoadedSize())
1034 {
1035 char szBuffer[200];
1036 snprintf(szBuffer, sizeof(szBuffer), "DebugDataCompare(): Compare size differs. %d compared to %d byte(s)", (int) _GetLoadedSize(), (int) pCompareObject->_GetLoadedSize());
1037 DebugOut(szBuffer);
1038 return;
1039 }
1040 unsigned char* pDataBlock = (unsigned char*) GetDataBlock();
1041 unsigned char* pCompareBlock = (unsigned char*) pCompareObject->GetDataBlock();
1042 int diffcount = 0;
1043 for (int i = 0; i < _GetLoadedSize(); i++)
1044 {
1045 if (pDataBlock[i] != pCompareBlock[i])
1046 {
1047 diffcount ++;
1048 char szBuffer[200];
1049 snprintf(szBuffer, sizeof(szBuffer), "DebugDataCompare() DIFF at 0-based byte offset %d: 0x%02x - 0x%02x ", i, pDataBlock[i], pCompareBlock[i]);
1050 DebugOut(szBuffer);
1051 }
1052 }
1053 if (diffcount == 0)
1054 {
1055 DebugOut("DebugDataCompare(): Data blocks are identical");
1056 }

2176 bool SplitAt(int splitpointindex, FCString* pFirstString, FCString* pSecondString, bool includesplitposchar) const
2177 {
2178 /* Make sure the 0-based index position is valid */
2179 if (splitpointindex < 0) return false;
2180 if (splitpointindex >= GetLength()) return false;
2181 if ((pFirstString) && (pFirstString == pSecondString))
2182 {
2183 /* Reference to same destination object is not allowed */
2184 return false;
2185 }
2186 if (pFirstString && pSecondString)
2187 {
2188 /* Don't allow copying to "this" if both substrings are used */
2189 if (pFirstString == this) return false;
2190 if (pSecondString == this) return false;
2191 }
2192 if (pFirstString)
2193 {
2194 pFirstString->SetString(this);
2195 /* Check state of includesplitposchar, to decide if the
2196 split point should be included or not. */
2197 int splitoffset = 0;
2198 if (includesplitposchar) splitoffset = 1;
2199 pFirstString->TruncateAt(splitpointindex + splitoffset);
2200 }
2201 if (pSecondString)
2202 {
2203 pSecondString->SetString(this);
2204 pSecondString->DeleteCharactersAt(0, splitpointindex + 1);
2205 }
2206 return true;
2207 }
