965 {
966 if (!pCompareObject)
967 {
968 DebugOut("DebugDataCompare(): Compare object is nil.");
969 return;
970 }
971 if (_GetLoadedSize() != pCompareObject->_GetLoadedSize())
972 {
973 char szBuffer[200];
974 sprintf(szBuffer, "DebugDataCompare(): Compare size differs. %d compared to %d byte(s)", (int) _GetLoadedSize(), (int) pCompareObject->_GetLoadedSize());
975 DebugOut(szBuffer);
976 return;
977 }
978 unsigned char* pDataBlock = (unsigned char*) GetDataBlock();
979 unsigned char* pCompareBlock = (unsigned char*) pCompareObject->GetDataBlock();
980 int diffcount = 0;
981 for (int i = 0; i < _GetLoadedSize(); i++)
982 {
983 if (pDataBlock[i] != pCompareBlock[i])
984 {
985 diffcount ++;
986 char szBuffer[200];
987 sprintf(szBuffer, "DebugDataCompare() DIFF at 0-based byte offset %d: 0x%02x - 0x%02x ", i, pDataBlock[i], pCompareBlock[i]);
988 DebugOut(szBuffer);
989 }
990 }
991 if (diffcount == 0)
992 {
993 DebugOut("DebugDataCompare(): Data blocks are identical");
994 }
995 }

1720 {
1721 /* Make sure the 0-based index position is valid */
1722 if (splitpointindex < 0) return false;
1723 if (splitpointindex >= GetLength()) return false;
1724 if ((pFirstString) && (pFirstString == pSecondString))
1725 {
1726 /* Reference to same destination object is not allowed */
1727 return false;
1728 }
1729 if (pFirstString && pSecondString)
1730 {
1731 /* Don't allow copying to "this" if both substrings are used */
1732 if (pFirstString == this) return false;
1733 if (pSecondString == this) return false;
1734 }
1735 if (pFirstString)
1736 {
1737 pFirstString->SetString(this);
1738 /* Check state of includesplitposchar, to decide if the
1739 split point should be included or not. */
1740 int splitoffset = 0;
1741 if (includesplitposchar) splitoffset = 1;
1742 pFirstString->TruncateAt(splitpointindex + splitoffset);
1743 }
1744 if (pSecondString)
1745 {
1746 pSecondString->SetString(this);
1747 pSecondString->DeleteCharactersAt(0, splitpointindex + 1);
1748 }
1749 return true;
1750 }
