1034 if (_GetLoadedSize() != pCompareObject->_GetLoadedSize())
1035 {
1036 char szBuffer[200];

1038 DebugOut(szBuffer);
1039 return;
1040 }
1041 unsigned char* pDataBlock = (unsigned char*) GetDataBlock();

1043 int diffcount = 0;
1044 for (int i = 0; i < _GetLoadedSize(); i++)
1045 {
1046 if (pDataBlock[i] != pCompareBlock[i])

1048 diffcount ++;
1049 char szBuffer[200];
1050 snprintf(szBuffer, sizeof(szBuffer), "DebugDataCompare() DIFF at 0-based byte offset %d: 0x%02x - 0x%02x ", i, pDataBlock[i], pCompareBlock[i]);
1051 DebugOut(szBuffer);
1052 }
1053 }
1054 if (diffcount == 0)
1055 {
1056 DebugOut("DebugDataCompare(): Data blocks are identical");

2236 {
2237 /* Make sure the 0-based index position is valid */
2238 if (splitpointindex < -1) return false;
2239 if (splitpointindex >= GetLength()) return false;
2240 if ((pFirstString) && (pFirstString == pSecondString))
2241 {
2242 /* Reference to same destination object is not allowed */
2243 return false;
2244 }
2245 if (pFirstString && pSecondString)
2246 {
2247 /* Don't allow copying to "this" if both substrings are used */
2248 if (pFirstString == this) return false;
2249 if (pSecondString == this) return false;
2250 }
2251 if (pFirstString)
2252 {
2253 pFirstString->SetString(this);
2254 /* Check state of includesplitposchar, to decide if the
2255 split point should be included or not. */
2256 const int splitoffset = includesplitposchar ? 1 : 0;
2257 // ignore return value here, because TruncateAt returns false if there is nothing to truncate,
2258 // but we do not care about that. We have already checked for the other conditions that might
2259 // cause TruncateAt to fail.
2260 pFirstString->TruncateAt((std::max)(0, splitpointindex + splitoffset));
2261 }
2262 if (pSecondString)
2263 {
2264 pSecondString->SetString(this);
2265 if (splitpointindex >= 0)
2266 pSecondString->DeleteCharactersAt(0, splitpointindex + 1);
2267 }
