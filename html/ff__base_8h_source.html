1027 {
1028 if (!pCompareObject)
1029 {
1030 DebugOut("DebugDataCompare(): Compare object is nil.");
1031 return;
1032 }
1033 if (_GetLoadedSize() != pCompareObject->_GetLoadedSize())
1034 {
1035 char szBuffer[200];
1036 snprintf(szBuffer, sizeof(szBuffer), "DebugDataCompare(): Compare size differs. %d compared to %d byte(s)", (int) _GetLoadedSize(), (int) pCompareObject->_GetLoadedSize());
1037 DebugOut(szBuffer);
1038 return;
1039 }
1040 unsigned char* pDataBlock = (unsigned char*) GetDataBlock();
1041 unsigned char* pCompareBlock = (unsigned char*) pCompareObject->GetDataBlock();
1042 int diffcount = 0;
1043 for (int i = 0; i < _GetLoadedSize(); i++)
1044 {
1045 if (pDataBlock[i] != pCompareBlock[i])
1046 {
1047 diffcount ++;
1048 char szBuffer[200];
1049 snprintf(szBuffer, sizeof(szBuffer), "DebugDataCompare() DIFF at 0-based byte offset %d: 0x%02x - 0x%02x ", i, pDataBlock[i], pCompareBlock[i]);
1050 DebugOut(szBuffer);
1051 }
1052 }
1053 if (diffcount == 0)
1054 {
1055 DebugOut("DebugDataCompare(): Data blocks are identical");
1056 }

2181 bool SplitAt(int splitpointindex, FCString* pFirstString, FCString* pSecondString, bool includesplitposchar) const
2182 {
2183 /* Make sure the 0-based index position is valid */
2184 if (splitpointindex < -1) return false;
2185 if (splitpointindex >= GetLength()) return false;
2186 if ((pFirstString) && (pFirstString == pSecondString))
2187 {
2188 /* Reference to same destination object is not allowed */
2189 return false;
2190 }
2191 if (pFirstString && pSecondString)
2192 {
2193 /* Don't allow copying to "this" if both substrings are used */
2194 if (pFirstString == this) return false;
2195 if (pSecondString == this) return false;
2196 }
2197 if (pFirstString)
2198 {
2199 pFirstString->SetString(this);
2200 /* Check state of includesplitposchar, to decide if the
2201 split point should be included or not. */
2202 const int splitoffset = includesplitposchar ? 1 : 0;
2203 // ignore return value here, because TruncateAt returns false if there is nothing to truncate,
2204 // but we do not care about that. We have already checked for the other conditions that might
2205 // cause TruncateAt to fail.
2206 pFirstString->TruncateAt((std::max)(0, splitpointindex + splitoffset));
2207 }
2208 if (pSecondString)
2209 {
2210 pSecondString->SetString(this);
2211 if (splitpointindex >= 0)
2212 pSecondString->DeleteCharactersAt(0, splitpointindex + 1);
2213 }
2214 return true;
2215 }
