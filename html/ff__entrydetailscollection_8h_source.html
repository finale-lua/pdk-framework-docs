177 FCBeamMod* GetOrCreateBeam(int beamno)
178 {
179 if (beamno < 2) return NULL;
180 if (beamno > 15) return NULL;
181 if (!_pNoteEntry)
182 {
183#ifdef PDK_FRAMEWORK_DIAGNOSE
184 DebugOut("No attached entry on FCSecondaryBeamMods::GetOrCreateBeam() call.");
185#endif
186 return NULL;
187
188 }
189 for (int i = 0; i < GetCount(); i++)
190 {
191 FCBeamMod* pBeamMod = (FCBeamMod*) GetItemAt(i);
192 if (pBeamMod->GetBeamNumber() == beamno) return pBeamMod;
193 }
194
195 FCBeamMod* pNewBeamMod = new FCBeamMod(true);
196 pNewBeamMod->SetNoteEntry(_pNoteEntry);
197 pNewBeamMod->SetBeamNumber(beamno);
198 pNewBeamMod->SaveNew();
199 Add(pNewBeamMod);
200 return pNewBeamMod;
201 }
202
212 FCBeamMod* GetBeam(int beamno)
213 {
214 for (int i = 0; i < GetCount(); i++)
215 {
216 FCBeamMod* pBeamMod = (FCBeamMod*) GetItemAt(i);
217 if (pBeamMod->GetBeamNumber() == beamno) return pBeamMod;
218 }
219 return NULL;
220 }
221
224 bool AllBeamsDataExists()
225 {
226 if (!_pNoteEntry) return false;
227
228 int beamcount = _pNoteEntry->CalcBeamCount();
229 for (int beamno = 2; beamno <= beamcount; beamno++)
230 {
231 bool found = false;
232 for (int i = 0; i < GetCount(); i++)
233 {
234 FCBeamMod* pBeamMod = (FCBeamMod*) GetItemAt(i);
235 if (pBeamMod->GetBeamNumber() == beamno) found = true;
236 }
237 if (!found) return false;
238 }
239 return true;
240 }
