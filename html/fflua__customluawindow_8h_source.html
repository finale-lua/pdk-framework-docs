182 bool RegisterHandleControlEvent(FCControl * pControl, luabridge::LuaRef lua_callback_function)

299 bool HandleControlEvent(FCControl* pControl)
300 {
301 auto it = _controlHandles.find(pControl);
302 if (it == _controlHandles.end()) return false;
303 bool result = false;
304 if (pControl && it->second.isFunction())
305 {
306 try
307 {
308 it->second(pControl);
309 result = true;
310 }
311 catch (luabridge::LuaException e)
312 {
313 _OutputExceptionError(e);
314 }
315 }
316 return result;
317 }
