154 eKey GetIDWithTransposition() const
155 {
156 /* Don't use other methods here, to avoid recursive calls.
157 * Instead, use the PDK helper functions and macros.
158 */
159 twobyte keyalteration = GetKeyAlteration(_key);
160
161 /* Chromatic key uses the standard key, so the transposition
162 alteration should NOT be set. */
163 keyalteration += _transposealteration;
164 if (_simplifykey)
165 {
166 if (keyalteration < -7)
167 {
168 keyalteration += 12;
169 }
170 if (keyalteration > 7)
171 {
172 keyalteration -= 12;
173 }
174 }
175
176
177 if (IsMinorKey(_key))
178 return MAKE_TWOBYTE( keyalteration, KEYMODE_MINOR );
179 else if (IsMajorKey(_key))
180 return MAKE_TWOBYTE( keyalteration, KEYMODE_MAJOR );
181 else
182 return _key;
183 }
