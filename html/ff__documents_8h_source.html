147 bool DisplayVisually()
148 {
149 /* Try to find the docid in a window first. */
150 int windowcount = FX_GetWindowCount();
151 if (windowcount == 0) return false;
152#if FXT_VERSION < FINALEVERSION_2012
153 int datasize = sizeof (EXDocWindow2009) * windowcount;
154 EXDocWindow2009* pDocwindowarray = new EXDocWindow2009[windowcount];
155#elif FXT_VERSION < FINALEVERSION_25
156 int datasize = sizeof (EXDocWindow2012) * windowcount;
157 EXDocWindow2012* pDocwindowarray = new EXDocWindow2012[windowcount];
158#else
159 int datasize = sizeof (EXDocWindow25) * windowcount;
160 EXDocWindow25* pDocwindowarray = new EXDocWindow25[windowcount];
161#endif
162 memset(pDocwindowarray, 0, datasize);
163 /* Don't look at the return value of FX_GetWindowInfoList,
164 * since it returns false on Finale 2012 at least. */
165 FX_GetWindowInfoList(pDocwindowarray, windowcount);
166 bool windowswitched = false;
167 for (int i = 0; i < windowcount; i++)
168 {
169 if (pDocwindowarray[i].docID == _docid)
170 {
171 if (!FX_SetCurrentWindow(pDocwindowarray[i].hWnd))
172 {
173 delete [] pDocwindowarray;
174 return false;
175 }
176 windowswitched = true;
177 break;
178 }
179 }
180 if (!windowswitched) {
181 delete [] pDocwindowarray;
182 return false;
183 }
184 /* Window have switched, make sure Enigma contents is correct as well. */
185 if (!FX_SetCurrentEnigmaDocument(_docid))
186 {
187 delete [] pDocwindowarray;
188 return false;
189 }
190 FX_RedrawMusic(NULL);
191 _lastdocid = 0;
192 return true;
193 }
